// 自律遊泳アルゴリズム（修正版） const velocity = new THREE.Vector3(0, 0, -0.01); const acceleration = new THREE.Vector3(); const dummy = new THREE.Object3D(); let swimTime = 0; // 尾びれのゆらゆら用タイマー function animate() { requestAnimationFrame(animate); if (fish && fish.object3D && fish.object3D.visible) { const obj = fish.object3D; swimTime += 0.05; // 1. ランダムな力（うろつき）を加える const wanderX = (Math.random() - 0.5) * 0.0006; const wanderY = (Math.random() - 0.5) * 0.0002; // 上下移動は少し控えめに const wanderZ = (Math.random() - 0.5) * 0.0006; acceleration.add(new THREE.Vector3(wanderX, wanderY, wanderZ)); // 2. 移動エリアの境界制限（離れすぎたら戻る） const dist = obj.position.length(); if (dist > 3.0) acceleration.add(obj.position.clone().normalize().multiplyScalar(-0.0005)); else if (dist < 1.0) acceleration.add(obj.position.clone().normalize().multiplyScalar(0.0005)); if (obj.position.y > 0.6) acceleration.y -= 0.0003; else if (obj.position.y < -0.6) acceleration.y += 0.0003; // 3. 速度の更新 velocity.add(acceleration); velocity.clampLength(0.005, 0.012); obj.position.add(velocity); acceleration.set(0, 0, 0); // ---------------------------------------------------- // 4. 自然な姿勢（向き）の計算 // ---------------------------------------------------- // (A) 左右の向き（Yaw） let yaw = Math.atan2(velocity.x, velocity.z); // (B) 上下の向き（Pitch）：傾きすぎないように±15度以内に制限！ const speedXZ = Math.sqrt(velocity.x * velocity.x + velocity.z * velocity.z); let pitch = Math.atan2(-velocity.y, speedXZ); const maxPitch = THREE.MathUtils.degToRad(15); // 最大15度 pitch = THREE.MathUtils.clamp(pitch, -maxPitch, maxPitch); // (C) 魚の「ゆらゆら」とした自然な泳ぎ（Sin波） const wiggle = Math.sin(swimTime) * 0.05; // (D) 3Dモデルの向き補正（元々のモデルの向きに合わせた90度オフセット） const modelOffset = Math.PI / 2; // Roll（Z軸＝お腹の傾き）を「0」に固定して回転を適用 dummy.rotation.set(pitch, yaw + modelOffset + wiggle, 0, 'YXZ'); // 滑らかに向きを追従させる（slerp） obj.quaternion.slerp(dummy.quaternion, 0.04); } } animate();
